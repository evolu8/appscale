521   # Queries Amazon EC2's Spot Instance pricing history to see how much other
 
522   # users have paid for the given instance type (assumed to be a Linux box),
 
523   # so that we can place a bid that is similar to the average price. How

527   def self.get_optimal_spot_price(instance_type)
 
528     command = "ec2-describe-spot-price-history -t #{instance_type} | " +
 
529       "grep 'Linux/UNIX' | awk '{print $2}'".split("\n")
 
530     prices = `#{command}`
 
531 
532     average = prices.reduce(0.0) { |sum, price|
 
533       sum += Float(price)
 
534     }

1031     return nil if input_string.nil? || input_string.empty?
