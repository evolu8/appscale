970   # Starts the given roles by using open nodes, spawning new nodes, or some
 
971   # combination of the two. 'nodes_needed' should be an Array, where each 
972   # item is an Array of the roles to start on each node.
 
973   def start_new_roles_on_nodes(nodes_needed, instance_type, secret)
 
974      if !valid_secret?(secret)
 
975       return BAD_SECRET_MSG
 
976     end
 
977 
978     if nodes_needed.class != Array
 
979       Djinn.log_debug("Was expecting nodes_needed to be an Array, not " +
 
980         "a #{nodes_needed.class}")
 
981       return BAD_INPUT_MSG
 
982     end
 
983 
984     Djinn.log_debug("Received a request to acquire nodes with roles " +
 
985       "#{nodes_needed.join(', ')}, with instance type #{instance_type} for " +
 
986       "new nodes")
 
987 
988     vms_to_use = []
 
989     ZKInterface.lock_and_run {
 
990       num_of_vms_needed = nodes_needed.length
 
991 
992       @nodes.each { |node|
 
993         if node.is_open?
 
994           Djinn.log_debug("Will use node #{node} to run new roles")
 
995           vms_to_use << node
 
996 
997           if vms_to_use.length == nodes_needed.length
 
998             Djinn.log_debug("Only using open nodes to run new roles")
 
999             break
 
1000           end
 
1001         end
 
1002       }
 
1003 
1004       vms_to_spawn = nodes_needed.length - vms_to_use.length
 
1005     
1006       if vms_to_spawn > 0 and !is_cloud?
 
1007         Djinn.log_debug("Still need #{vms_to_spawn} more nodes, but we " +
 
1008         "aren't in a cloud environment, so we can't acquire more nodes - " +
 
1009         "failing the caller's request.")
 
1010         return NOT_ENOUGH_OPEN_NODES
 
1011       end
 
1012 
1013       if vms_to_spawn > 0
 
1014         Djinn.log_debug("Need to spawn up #{vms_to_spawn} VMs")
 
1015         # start up vms_to_spawn vms as open
 
1016         imc = InfrastructureManagerClient.new(@@secret)
 
1017         new_nodes_info = imc.spawn_vms(vms_to_spawn, @creds, "open", "cloud1")
 
1018 
1019         # initialize them and wait for them to start up
 
1020         Djinn.log_debug("info about new nodes is " +
 
1021           "[#{new_nodes_info.join(', ')}]")
 
1022         add_nodes(new_nodes_info) 
1023 
1024         # add information about the VMs we spawned to our list, which may
 
1025         # already have info about the open nodes we want to use
 
1026         new_nodes = Djinn.convert_location_array_to_class(new_nodes_info,
 
1027           @creds['keyname'])
 
1028         vms_to_use << new_nodes
 
1029         vms_to_use.flatten!
 
1030       end
 
1031     }
 
1032 
1033     wait_for_nodes_to_finish_loading(vms_to_use)
 
1034     
1035     nodes_needed.each_index { |i|
 
1036       Djinn.log_debug("Adding roles #{nodes_needed[i].join(', ')} " +
 
1037         "to virtual machine #{vms_to_use[i]}")
 
1038       ZKInterface.add_roles_to_node(nodes_needed[i], vms_to_use[i])
 
1039     }
 
1040 
1041     wait_for_nodes_to_finish_loading(vms_to_use)
 
1042 
1043     return "OK"
 
1044   end

1701   def update_local_nodes
 
1702     Djinn.log_debug("Getting ZK lock to update @nodes")
 
1703 
1704     ZKInterface.lock_and_run {
 
1705       # See if the ZooKeeper data is newer than ours - if not, don't
 
1706       # update anything and return.
 
1707       zk_ips_info = ZKInterface.get_ip_info()
 
1708       if zk_ips_info["last_updated"] <= @last_updated
 
1709         Djinn.log_debug("Latest ZooKeeper data does not have newer data " +
 
1710           "than us. ZK timestamp is #{zk_ips_info['last_updated']}, our" +
 
1711           " timestamp  is #{@last_updated}")
 
1712         return "NOT UPDATED"
 
1713       else
 
1714         Djinn.log_debug("Updating data from ZK. Our timestamp, " +
 
1715           "#{@last_updated}, was older than the ZK timestamp, " +
 
1716           "#{zk_ips_info['last_updated']}")
 
1717       end
 
1718 
1719       all_ips = zk_ips_info["ips"]
 
1720       new_nodes = []
 
1721       all_ips.each { |ip|
 
1722         new_nodes << DjinnJobData.deserialize(
 
1723           ZKInterface.get_job_data_for_ip(ip))
 
1724       }
 
1725 
1726       old_roles = my_node.jobs
 
1727       @nodes = new_nodes
 
1728       find_me_in_locations
 
1729       new_roles = my_node.jobs
 
1730 
1731       Djinn.log_debug("My new nodes are [#{@nodes.join(', ')}], and my new " +
 
1732         "node is #{my_node}")
 
1733 
1734       # Since we're about to possibly load and unload roles, set done_loading
 
1735       # for our node to false, so that other nodes don't erroneously send us
 
1736       # additional roles to do while we're in this state where lots of side
 
1737       # effects are happening.
 
1738       @done_loading = false
 
1739       ZKInterface.set_done_loading(my_node.public_ip, false)
 
1740    
1741       roles_to_start = new_roles - old_roles
 
1742       if !roles_to_start.empty?
